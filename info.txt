# Перейти в каталог проекта
cd /path/to/your/project

# Инициализировать репозиторий (если еще не инициализирован)
git init

# Клонировать репозиторий (если еще не клонировали)
git clone https://github.com/BigManJunior/game.git

# Перейти в клонированный репозиторий
cd game

# Проверить статус
git status

# Добавить изменения в индекс
git add .

# Закоммитить изменения
git commit -m "Your commit message"

# Синхронизировать с удаленным репозиторием
git pull origin main

# Отправить изменения в удаленный репозиторий
git push origin main
